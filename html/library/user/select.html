【引用来自layui扩展平台的多选和级联扩展组件举例】

larryms框架模板中支持第三方自定义符合layui模板定义规范的组件，如layui社区扩展平台分享的select组【larryms官方不提供当前自定义示例的组件维护，只是作为layui扩展平台下的组件在larryms框架模板的使用支持引入举例之一，larryms授权用户可以参看该示例介绍 使用其他自定义扩展组件，或文档中的关于自定义扩展组件的使用详细介绍】larryms中对用户自行加入的扩展组件支持良好，无需额外使用别名引入或页面手动引入 就像使用layui自带组件一样使用，真正的开箱即用，按需加载

//以下是部分代码是第三方组件在larryms中的使用举例，具体可查看demo源码中的详细示例

      var $ = layui.$,
          form = layui.form,
          selectN = layui.selectN,
          selectM = layui.selectM;
         //这里的json数据实际项目中可以是来自后端接口请求返回，下面是layui扩展平台中分享的组件示例demo
      var catData = [
          {"id":1,"name":"周边旅游","children":[{"id":24,"name":"广东","status":0,"children":[{"id":7,"name":"广州"},{"id":23,"name":"潮州"}]}]},{"id":5,"name":"国内旅游","children":[{"id":8,"name":"华北地区","children":[{"id":9,"name":"北京"}]}]},{"id":6,"name":"出境旅游","children":[{"id":10,"name":"东南亚","children":[{"id":11,"name":"马来西亚","children":[{"id":20,"name":"沙巴","children":[{"id":21,"name":"美人鱼岛","children":[{"id":22,"name":"潜水"}]}]}]}]}]}];
          //无限级分类-基本配置
      var catIns1 = selectN({//元素容器【必填】
            elem: '#cat_ids1',
            search: [false, true],//候选数据【必填】
            data: catData
      });
});
